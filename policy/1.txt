    function changeColor(selector) {

            selector.style.color = '#000';
        }
        var catid = $(selector).val();

    var loading = false;
    var count = 10;  //初始展示记录数
    var total = "1176";  //总记录数
    var loaded = false; //加载完成flag
    var page = 1; //页码
    if(total > count) {
        $(document.body).infinite().on("infinite", function() {
            if(loading | loaded) return;
            loading = true;
            var load_num = 10; //每次加载记录数
            if(total - count < 10) {
                load_num = total - count;
            }
            get_data(load_num);
        });
    }else{
        $('.weui-loadmore').hide();
    }

    function get_data(load_num,is_form,catid) {
        var is_catid = catid ? catid : "41";
        page++;
        if(is_form){
            var url = "m=content&c=index&a=policy_lists&dosubmit=1&catid="+is_catid;
        }else{
            var url = "m=content&c=index&a=policy_lists&dosubmit=1&catid="+is_catid+"&page=" + page;
        }
        $.ajax({
            type: "GET",
            url: "index.php",
            data: url,
            dataType: 'json',
            success: function(data) {
                if (data.Code == 0) {
                    var reslength = data.Data.length;
                    var res = new Array();
                    if (reslength > 0) {
                        if(is_form){
                            $("#info-list").html('');
                        }
                        for(var i=0; i<reslength; i++) {
                            res = $.makeArray(data.Data[i]);
                            $("#info-list").append('<a href="'+res[0].url+'" class="weui-media-box weui-media-box_appmsg show-loading">'+
                                '<div class="weui-media-box_text">'+
                                '<h4 class="weui-media-box__title">'+ res[0].title.substring(0,20) +'...</h4>'+
                                '<p class="weui-media-box__desc">'+ res[0].description.substring(0,65) +'...</p>' +
                                '</div>' +
                                '</a>');
                            res = new Array();
                        }
                        $('.weui-loadmore').html('<span class="weui-loadmore__tips">加载完成</span>');
                    } else {
                        if(is_form){
                            $("#info-list").html('<a style="padding: 20px 140px;" href="javascript:void(0)" class="weui-media-box weui-media-box_appmsg show-loading">'+
                                '<div class="weui-media-box_text">'+
                                '<h4 class="weui-media-box__title">暂无数据</h4>'+
                                '</div>' +
                                '</a>');
                        }else{
                            $('.weui-loadmore').html('<span class="weui-loadmore__tips">加载完成</span>');
                        }
                    }
                }
                $('.weui-loadmore').hide();
            }
        });
        loading = false;
        count += load_num;
    }
